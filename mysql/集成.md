创建程序专用账号

```shell
create user 'service'@'%' identified WITH mysql_native_password by 'aaaaaaaaaa';
GRANT SELECT, INSERT, UPDATE, DELETE, CREATE, DROP, RELOAD, PROCESS, REFERENCES, INDEX, ALTER, SHOW DATABASES, CREATE TEMPORARY TABLES, LOCK TABLES, EXECUTE, REPLICATION SLAVE, REPLICATION CLIENT, CREATE VIEW, SHOW VIEW, CREATE ROUTINE, ALTER ROUTINE, CREATE USER, EVENT, TRIGGER ON *.* TO `service`@`%` WITH GRANT OPTION;
#REVOKE INSERT, UPDATE, DELETE, CREATE, DROP, INDEX, ALTER ON `mysql`.* FROM `service`@`%`;
flush privileges;



create user 'debug_r'@'%' identified WITH mysql_native_password by 'aaaaaaaaaa';
GRANT SELECT, REFERENCES, INDEX, SHOW DATABASES ON *.* TO `debug_r`@`%` WITH GRANT OPTION;
flush privileges;


create user 'dev_rw'@'%' identified WITH mysql_native_password by 'aaaaaaaaaa';
GRANT SELECT, INSERT, UPDATE, DELETE, CREATE, DROP, RELOAD, PROCESS, REFERENCES, INDEX, ALTER, SHOW DATABASES, CREATE TEMPORARY TABLES, LOCK TABLES, EXECUTE, REPLICATION SLAVE, REPLICATION CLIENT, CREATE VIEW, SHOW VIEW, CREATE ROUTINE, ALTER ROUTINE, CREATE USER, EVENT, TRIGGER ON *.* TO `dev_rw`@`%` WITH GRANT OPTION;
#REVOKE INSERT, UPDATE, DELETE, CREATE, DROP, INDEX, ALTER ON `mysql`.* FROM `service`@`%`;
flush privileges;
```
